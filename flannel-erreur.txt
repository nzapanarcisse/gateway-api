exécuter ces commandes sur all noeud


sudo mkdir -p /run/flannel
sudo tee /run/flannel/subnet.env <<EOF
FLANNEL_NETWORK=10.244.0.0/16
FLANNEL_SUBNET=10.244.0.1/24
FLANNEL_MTU=1450
FLANNEL_IPMASQ=true
EOF


sudo modprobe br_netfilter
sudo bash -c 'cat <<EOF > /etc/modules-load.d/k8s.conf
br_netfilter
overlay
EOF'

sudo bash -c 'cat <<EOF > /etc/sysctl.d/k8s.conf
net.bridge.bridge-nf-call-iptables  = 1
net.bridge.bridge-nf-call-ip6tables = 1
net.ipv4.ip_forward                 = 1
EOF'

# Appliquer les paramètres sans redémarrer
sudo sysctl --system

kubectl delete pod kube-flannel-ds-bs7sc -n kube-flannel
